\section{python使用笔记}
\subsection{python计算器}
\begin{itemize}
\item 幂运算
\begin{commandbox}
>>> 2**16 - 1
65535
\end{commandbox}

\item 十六进制转换运算
\begin{commandbox}
>>> hex(16)
'0x10'
\end{commandbox}
\item 二进制转换运算
\begin{commandbox}
>>> bin(0xfe)
'0b11111110'
\end{commandbox}
\end{itemize}
